import nodes from "./supported.json";
import allNodes from "./all.json";

# Supported Python Syntax

export const Node = ({ name }) => {
  const ok = nodes.includes(name);
  return (
    <li>
      <strong>
        {ok ? (
          "✅"
        ) : (
"❌"
        )}{" "}
        ast.{name}
      </strong>
      {" - "}
      <a href={"https://docs.python.org/3.9/library/ast.html#ast." + name}>
        {" "}
        {"https://docs.python.org/3.9/library/ast.html#ast." + name}{" "}
      </a>
    </li>
  );
};

export const NodeList = ({ nodes }) => {
  return nodes.map((x) => <Node name={x}></Node>);
};

<ul className="nodelist">
  <NodeList nodes={allNodes}></NodeList>
</ul>
