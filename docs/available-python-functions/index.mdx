import docs from "./dyslang_docs.json";
import styles from "./styles.css";

export const DocList = ({ docs }) => {
  console.log('DocList')
  toc.length = 0;
  return Object.keys(docs).map((item, i) => {
    console.log(item)
    const d = docs[item];
    toc.push({ value: item, id: item, level: 2 });
    //const [lastmod, name] = toc[toc.length - 1].split("/");
    return (
      <dl key={i} className="doc" id={item}>
        <dt>
          <strong>
            <a href={"#" + item}>
              {item}
              {d.init_definition}
              {d.definition}
            </a>
          </strong>
        </dt>
        <dl>
          <pre>{d.docstring}</pre>
        </dl>
      </dl>
    );
  });
};

<div className="doclist">
  <DocList docs={docs}></DocList>
</div>
